def compute
  case provider_name
  when "aws"
    @compute ||= Fog::Compute.new(:provider => "AWS", 
      :aws_access_key_id => account_key, 
      :aws_secret_access_key => account_secret,
      :region => region)
  else
    error("Provider '#{provider_name}' is currently unsupported.")
  end
end
